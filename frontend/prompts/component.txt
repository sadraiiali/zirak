                    if (diffMinutes < 1) return `${diffSeconds} seconds ago`;
                    if (diffMinutes < 60) return `${diffMinutes} minutes ago`;
                    return date.toLocaleString([], { dateStyle: 'short', timeStyle: 'short' });
